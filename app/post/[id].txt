import { View, Text, Image, StyleSheet, ScrollView, Dimensions, TouchableOpacity, ActivityIndicator } from 'react-native';
import React, { useState, useEffect } from 'react';
import PostService, { type Post } from '../../services/PostService';

export default function PostDetailScreen() {
  const { id } = useLocalSearchParams();
  const router = useRouter();
  const [post, setPost] = useState<Post | null>(null);
  const [loading, setLoading] = useState(true);
  const [activeIndex, setActiveIndex] = useState(0);

  useEffect(() => {
    loadPost();
  }, [id]);

  const loadPost = async () => {
    try {
      setLoading(true);
      const result = await PostService.getPost(parseInt(id as string));
      if (result.success) {
        setPost(result.post);
      }
    } catch (error) {
      console.error('ê²Œì‹œê¸€ ë¡œë”© ì‹¤íŒ¨:', error);
    } finally {
      setLoading(false);
    }
  };

  if (loading) {
    return (
      <View style={[styles.container, styles.centerContent]}>
        <ActivityIndicator size="large" color="#3B82F6" />
        <Text style={styles.loadingText}>ê²Œì‹œê¸€ì„ ë¶ˆëŸ¬ì˜¤ëŠ” ì¤‘...</Text>
      </View>
    );
  }

  if (!post) {
    return (
      <View style={[styles.container, styles.centerContent]}>
        <Text style={styles.errorText}>ê²Œì‹œê¸€ì„ ì°¾ì„ ìˆ˜ ì—†ìŠµë‹ˆë‹¤.</Text>
      </View>
    );
  }

  const screenWidth = Dimensions.get('window').width;

  // ì´ë¯¸ì§€ ê²½ë¡œ ì²˜ë¦¬ í•¨ìˆ˜
  const getImageSource = (imagePath: string) => {
    if (imagePath.startsWith('http')) {
      return { uri: imagePath };
    }
    // ì„œë²„ì˜ ì—…ë¡œë“œëœ ì´ë¯¸ì§€ ê²½ë¡œ
    return { uri: `http://10.0.2.2:3000${imagePath}` };
  };

  return (
    <ScrollView style={styles.container}>
      {/* í—¤ë” */}
      <View style={styles.postHeader}>
        <Image
          source={{ uri: 'https://via.placeholder.com/40' }}
          style={styles.profileImage}
        />
        <View style={styles.headerText}>
          <Text style={styles.userName}>{post.userName}</Text>
          <Text style={styles.postTime}>
            {new Date(post.createdDate).toLocaleDateString('ko-KR')} â€¢ ì¡°íšŒ {post.viewCount}
          </Text>
        </View>
      </View>

      {/* ì œëª© (ìˆëŠ” ê²½ìš°) */}
      {post.title && (
        <Text style={styles.postTitle}>{post.title}</Text>
      )}

      {/* ì´ë¯¸ì§€ ìŠ¬ë¼ì´ë” */}
      {post.imagePaths && post.imagePaths.length > 0 && (
        <>
          <ScrollView
            horizontal
            pagingEnabled
            onScroll={(e) => {
              const idx = Math.round(e.nativeEvent.contentOffset.x / screenWidth);
              setActiveIndex(idx);
            }}
            scrollEventThrottle={16}
            style={styles.imageSlider}
          >
            {post.imagePaths.map((img, idx) => (
              <Image
                key={idx}
                source={getImageSource(img)}
                style={[styles.postImage, { width: screenWidth - 32 }]}
              />
            ))}
          </ScrollView>

          {/* ì¸ë””ì¼€ì´í„° */}
          {post.imagePaths.length > 1 && (
            <View style={styles.indicatorContainer}>
              {post.imagePaths.map((_, idx) => (
                <View
                  key={idx}
                  style={[
                    styles.indicator,
                    activeIndex === idx && styles.indicatorActive,
                  ]}
                />
              ))}
            </View>
          )}
        </>
      )}

      {/* ë‚´ìš© */}
      <Text style={styles.postContent}>{post.content}</Text>

      {/* ê²½ë¡œ ì •ë³´ */}
      {post.pathName && (
        <View style={styles.pathInfoContainer}>
          <Text style={styles.pathInfoTitle}>ğŸ“ ì—°ê²°ëœ ì‚°ì±…ë¡œ</Text>
          <Text style={styles.pathInfoText}>{post.pathName}</Text>
        </View>
      )}

      {/* ë²„íŠ¼ 2ê°œ */}
      <View style={styles.buttonRow}>
        <TouchableOpacity
          style={[styles.actionButton, styles.saveButton]}
          onPress={() => router.push(`/post/${id}/save-path`)}
        >
          <Text style={styles.buttonText}>ê²½ë¡œ ì €ì¥</Text>
        </TouchableOpacity>
        
        <TouchableOpacity
          style={[styles.actionButton, styles.navigateButton]}
          onPress={() => router.push(`/post/${id}/navigate-path`)}
        >
          <Text style={styles.buttonText}>ê²½ë¡œ ì•ˆë‚´</Text>
        </TouchableOpacity>
      </View>
    </ScrollView>
  );
}
